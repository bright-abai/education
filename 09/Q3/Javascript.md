```javascript
var input = require('fs').readFileSync('/dev/stdin', 'utf8');
var lines = input.split('\n');

const PI = 3.14159;
let radius = parseFloat(lines[0]);
let area = radius * radius * PI; 

console.log("A=" + area.toFixed(4));
```
